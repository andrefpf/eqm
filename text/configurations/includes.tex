\newcommand*\trackglsusage[1]{%
  \begingroup%
    \edef\tempglabel{#1}%
    \edef\tempglstype{\glsentrytype{\tempglabel}}%
    % \typeout{DEBUG: Using label = [\tempglabel], type = [\tempglstype]}%
    \ifx\tempglstype\acronymtype%
      \stepcounter{acronymusecount}%
      % \typeout{DEBUG: Counted as acronym.}%
    \else%
      % \typeout{DEBUG: Not counted as acronym.}%
      \ifx\tempglstype\notationtype%
        % \typeout{DEBUG: Counted as notation.}%
      \else%
        \stepcounter{symbolusecount}%
        % \typeout{DEBUG: Counted as symbol.}%
      \fi%
    \fi%
  \endgroup%
}
