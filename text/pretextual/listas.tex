%%% LISTA DE ACRÔNIMOS
\newcommand{\inserirListaDeAcronimos}{%
  \pdfbookmark[0]{\listadesiglasname}{loc}%
  \renewcommand*\glspostdescription{\dotfill}
  \setlist[description]{leftmargin=!, labelwidth=5em}
  \printglossary[type=\acronymtype,title={\listadesiglasname}]%
  \setlist[description]{style=standard}
  \cleardoublepage%  
}%

%%% LISTA DE SÍMBOLOS
\newcommand{\inserirListaDeSimbolos}{%
  \pdfbookmark[0]{\listadesimbolosname}{los}%
  %\PRIVATEbookmarkthis{\listadesimbolosname}
  \renewcommand*\glspostdescription{\dotfill}
  \setlist[description]{leftmargin=!, labelwidth=3em}
  \newcounter{dummySymbol}
  \refstepcounter{dummySymbol}
  \label{list.symbols}
  \printglossary[type=symbols,title={\listadesimbolosname}, nogroupskip]%
  \setlist[description]{style=standard}
  \cleardoublepage%  \printglossary[type=symbols, title={\listadesimbolosname}, nogroupskip]
}%

\newcommand{\insertListOfSymbols}{\inserirListaDeSimbolos{}}%

\newcommand{\inserirNotacao}{%

  %\PRIVATEbookmarkthis{\listadesimbolosname}

  \refstepcounter{dummyNotation}
  \label{list.notation}
  \renewcommand*\glspostdescription{}
  \printglossary[type=notation,title={\notacaoname},style=tree,nonumberlist]%
  \cleardoublepage%  
}%
